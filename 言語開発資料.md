## 関数

```C𢌽

// 戻り値無しの関数
func void function(){  
    //処理
}

// 戻り値有りの関数
func int function(){  
    //処理
    return 0;
}

// 戻り値の指定が自由な（Pythonのよう）な関数
func function(){  
    //処理
    if (条件) {
        //条件が真の場合
        return "OK";
    } else {
        //条件が偽の場合
        return 0;
    }
}
```
